\begin{exercise}
Which fraction is larger?
\[
\frac{14}{87} \quad \text{or} \quad \frac{16}{89}
\]
\begin{prompt}
\[
\answer{\frac{14}{87}} < \answer{\frac{16}{89}}
\]
\end{prompt}
\begin{exercise}
Jon thinks that $\frac{14}{87} < \frac{16}{89}$ because $14 < 16$.  Select the most correct
response below.
\begin{multipleChoice}
\choice{Jon is completely correct.}
\choice[correct]{Jon is correct that $14 < 16$, but comparing numerators is not enough information to compare fractions.}
\choice{Jon is correct that $\frac{14}{87} < \frac{16}{89}$, but the reason is that $87 < 89$.}
\choice{Jon is completely incorrect.}
\end{multipleChoice}
\end{exercise}
\end{exercise}
